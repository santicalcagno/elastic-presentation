do
	for b in "${B_VALUES[@]}"
	do
		sed -i -e "s/-w [0-9]*/-w $w/" docker-compose.yml
		sed -i -e "s/-b [0-9]*/-b $b/" docker-compose.yml
		docker-compose up &
		DOCKER_PID=$!
		sh ./gatherdata.sh &
		GATHER_PID=$!
		python jlog.py
		kill $GATHER_PID &&
		curl -s -XDELETE 'http://localhost:9200/_all'
		kill $DOCKER_PID
	done
done
